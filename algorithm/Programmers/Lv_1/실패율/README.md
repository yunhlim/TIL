def solution(N, stages):

    for stage in stages:
        stage_cnt_dict[stage] = stage_cnt_dict.get(stage, 0) + 1

    arr_2 = []

        if arr[i][0] == 0:
            arr_2.append([0, i])

    arr_2.sort(key=lambda x: [-x[0], x[1]])

    answer = []
    for v in arr_2:
        answer.append(v[1])
    
    return answer
