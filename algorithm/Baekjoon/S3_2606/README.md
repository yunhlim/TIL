from collections import deque


n = int(input())    # μ»΄ν“¨ν„° μ
e = int(input())    # κ°„μ„ μ μ
arr = [[] for _ in range(n + 1)]   # μ •μ λ“¤μ μ—°κ²°κ΄€κ³„

        visited[v] = 1      # ν™•μΈν–μμ„ ν‘μ‹
        for v2 in arr[v]:   # μ—°κ²°λμ–΄μλ” μ»΄ν“¨ν„°λ¥Ό νμ— λ‹΄μ•„μ¤€λ‹¤.
