## ğŸ“’ BFS ì½”ë“œ

```python
from collections import deque


n = int(input())
arr = [0] * (n + 1)
queue = deque()
queue.append(1)

while queue and arr[n] == 0:
    v = queue.popleft()
    if v + 1 < len(arr) and arr[v+1] == 0:
        arr[v+1] = arr[v] + 1
        queue.append(v + 1)
    if v * 2 < len(arr) and arr[v*2] == 0:
        arr[v*2] = arr[v] + 1
        queue.append(v * 2)
    if v * 3 < len(arr) and arr[v*3] == 0:
        arr[v*3] = arr[v] + 1
        queue.append(v * 3)
print(arr[n])
```
