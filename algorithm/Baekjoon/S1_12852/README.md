from collections import deque


def que_add(node, nxt):         # ν™•μΈ ν›„ visited μ—…λ°μ΄νΈ λ° νμ— λ‹΄κΈ°
    if in_range(nxt) and visited[nxt] == []:    # λ²”μ„ μ²΄ν¬, λ°©λ¬Έ ν™•μΈ
        visited[nxt] = visited[node] + [nxt]    # λ°©λ¬Έν•μ§€ μ•μ•μΌλ©΄ μ΄μ „ κ°’λ“¤μ„ λ„£μ–΄μ¤€λ‹¤.
        queue.append(nxt)                       # νμ— λ„£λ”λ‹¤.


def in_range(x):                # λ²”μ„ μ²΄ν¬
    return 0 < x <= 1000000


n = int(input())
visited = [[] for _ in range(1000001)]      # ν™•μΈν• κ°’λ“¤μ„ λ„£μ–΄μ¤€λ‹¤.
mul = [3, 2]    # 3, 2 λ‚λ„κΈ°

queue = deque()
queue.append(n)

while queue:    # BFS
    node = queue.popleft()
    if node == 1:           # nμ΄ λ‚μ™”μΌλ©΄ μΆ…λ£
        break
    for j in range(2):      # λ‚λ„κΈ°
        if node % mul[j] == 0:  # λ‚λ„μ–΄ λ–¨μ–΄μ§€λ” κ²½μ°μ—λ§
            nxt = node // mul[j]    # λ‚λλ‹¤.
            que_add(node, nxt)
    # -1
    nxt = node - 1
    que_add(node, nxt)

print(len(visited[1]) - 1)
