```python
import sys
input = sys.stdin.readline

n, m = map(int, input().split())
arr = [0] + list(map(int, input().split()))

memo = [[arr[i]] for i in range(n + 1)]

for _ in range(m):
    i, j = map(int, input().split())
    while len(memo[i]) <= j - i:
        memo[i].append(memo[i][len(memo[i]) - 1] + arr[len(memo[i]) + i])

    print(memo[i][j-i])

```
